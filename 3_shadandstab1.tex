    \item \label{item:shadimplsolv}  Shadowing \imply solvability of system of linear difference equations
    $$
    x_{k+1} = A_k x_k + f_{k+1},\qquad k\in \Zb.
      $$
      Bounded $\seqz{f}{k}$ \imply bounded solution $\seqz{x}{k}$

\end{enumerate}
